    \textbf{หมายเหตุ}\; binary operator \lstinline{mod} 
    คือการหารที่เอาเฉพาะเศษเป็นผลลัพธ์ของนิพจน์นั้น เช่น \lstinline{13 mod 3 = 1}
}

    \lstinline{((rand7() + rand7()) mod 11) + 1}
\end{center}
